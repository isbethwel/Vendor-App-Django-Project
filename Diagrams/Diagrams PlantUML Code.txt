//Flowchart
@startuml 
Title Flowchart for Document Search Process
start
:Enter Search Query;
if (Valid query?) then (yes)
  :Query Database for Documents;
  if (Documents Found?) then (yes)
    :Display Documents;
  else (no)
    :Display "No Documents Found";
  endif
else (no)
  :Prompt for a Valid Query;
endif
stop
@enduml
